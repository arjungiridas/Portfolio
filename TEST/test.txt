// DARK MODE
function toggleMode() {
    document.body.classList.toggle("dark");
}

// PARALLAX
document.addEventListener("mousemove", e => {
    document.getElementById("layer1").style.transform =
        `translate(${e.clientX/20}px, ${e.clientY/20}px)`;

    document.getElementById("layer2").style.transform =
        `translate(${e.clientX/40}px, ${e.clientY/40}px)`;
});

// 3D TILT
document.querySelectorAll(".tilt").forEach(card => {
    card.addEventListener("mousemove", e => {
        let x = e.offsetX;
        let y = e.offsetY;
        let rotateX = -(y - 100) / 10;
        let rotateY = (x - 150) / 10;

        card.style.transform = `rotateX(${rotateX}deg) rotateY(${rotateY}deg)`;
    });

    card.addEventListener("mouseleave", () => {
        card.style.transform = "rotateX(0) rotateY(0)";
    });
});
